{
  const syn = synthetic[selected_forecast];
  const pums_vals = pums.income;
  const labels = income_labels;
  const trace_pums = {
    x: labels,
    y: pums_vals,
    name: "PUMS " + base_year,
    type: "bar",
    marker: { color: "steelblue" }
  };
  const trace_syn = {
    x: labels,
    y: syn.income,
    name: "Synthetic " + forecast_year,
    type: "bar",
    marker: { color: "darkorange" }
  };
  const layout = {
    barmode: "group",
    yaxis: { title: "Proportion of Households", tickformat: ".1%" },
    legend: { orientation: "h", y: 1.12 },
    margin: { t: 30, b: 80 }
  };
  const div = DOM.element("div");
  Plotly.newPlot(div, [trace_pums, trace_syn], layout, { responsive: true });
  return div;
}
